// EXEMPLE 2: Embedding avec Filtres Dynamiques (React)
import React, { useState } from 'react';

function SalesDashboard() {
  const [dateRange, setDateRange] = useState('7d');

  // Base embed URL
  const baseUrl = "https://charts.mongodb.com/charts-project-xxxxx/embed/charts?id=648f1234-abcd-1234-5678-9abcdef01234";

  // Ajouter filtres dynamiques
  const filter = {
    createdAt: {
      $gte: getDateFromRange(dateRange)
    }
  };

  const embedUrl = `${baseUrl}&filter=${encodeURIComponent(JSON.stringify(filter))}`;

  return (
    <div>
      <div>
        <button onClick={() => setDateRange('7d')}>Last 7 Days</button>
        <button onClick={() => setDateRange('30d')}>Last 30 Days</button>
        <button onClick={() => setDateRange('90d')}>Last 90 Days</button>
      </div>

      <iframe
        src={embedUrl}
        width="100%"
        height="600"
        frameBorder="0"
      />
    </div>
  );
}

function getDateFromRange(range) {
  const now = new Date();
  const days = parseInt(range);
  return new Date(now.setDate(now.getDate() - days));
}
